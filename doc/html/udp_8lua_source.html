176 --- Resolve which header comes after this one (in a packet).
177 --- For instance: in tcp/udp based on the ports.
178 --- This function must exist and is only used when get/dump is executed on
179 --- an unknown (mbuf not yet casted to e.g. tcpv6 packet) packet (mbuf)
180 --- @return String next header (e.g. 'udp', 'icmp', nil)
181 function udpHeader:resolveNextHeader()
182  local port = self:getDstPort()
183  for name, _port in pairs(mapNamePort) do
184  if type(_port) == "table" then
185  for _, p in pairs(_port) do
186  if port== p then
187  return name
188  end
189  end
190  elseif port == _port then
191  return name
192  end
193  end
194  return nil
195 end

197 --- Change the default values for namedArguments (for fill/get).
198 --- This can be used to for instance calculate a length value based on the total packet length.
199 --- See proto/ip4.setDefaultNamedArgs as an example.
200 --- This function must exist and is only used by packet.fill.
201 --- @param pre The prefix used for the namedArgs, e.g. 'udp'
202 --- @param namedArgs Table of named arguments (see See Also)
203 --- @param nextHeader The header following after this header in a packet
204 --- @param accumulatedLength The so far accumulated length for previous headers in a packet
205 --- @see ip4Header:fill
206 function udpHeader:setDefaultNamedArgs(pre, namedArgs, nextHeader, accumulatedLength)
207  -- set length
208  if not namedArgs[pre .. "Length"] and namedArgs["pktLength"] then
209  namedArgs[pre .. "Length"] = namedArgs["pktLength"] - accumulatedLength
210  end
211 
212  -- set dst port
213  if not namedArgs[pre .. "Dst"] then
214  for name, _port in pairs(mapNamePort) do
215  if nextHeader == name then
216  namedArgs[pre .. "Dst"] = type(_port) == "table" and _port[1] or _port
217  break
218  end
219  end
220  end
221  return namedArgs
222 end
