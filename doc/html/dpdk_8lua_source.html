32 --- Inits DPDK. Called by MoonGen on startup.
33 function mod.init()
34  -- register drivers
35  dpdkc.register_pmd_drivers()
36  -- TODO: support arbitrary dpdk configurations by allowing configuration in the form ["cmdLine"] = "foo"
37  local cfgFileLocations = {
38  "./dpdk-conf.lua",
39  "./lua/dpdk-conf.lua",
40  "../lua/dpdk-conf.lua",
41  "/etc/moongen/dpdk-conf.lua"
42  }
43  local cfg
44  for _, f in ipairs(cfgFileLocations) do
45  if fileExists(f) then
46  cfgScript = loadfile(f)
47  setfenv(cfgScript, setmetatable({ DPDKConfig = function(arg) cfg = arg end }, { __index = _G }))
48  local ok, err = pcall(cfgScript)
49  if not ok then
50  print("could not load DPDK config: " .. err)
51  return false
52  end
53  if not cfg then
54  print("config file does not contain DPDKConfig statement")
55  return false
56  end
57  cfg.name = f
58  break
59  end
60  end
61  if not cfg then
62  print("no DPDK config found, using defaults")
63  cfg = {}
64  end
65  local coreMask
66  if not cfg.cores then
67  -- default: use all the cores
68  local cpus = io.open("/proc/cpuinfo", "r")
69  cfg.cores = {}
70  for cpu in cpus:read("*a"):gmatch("processor : (%d+)") do
71  cfg.cores[#cfg.cores + 1] = tonumber(cpu)
72  end
73  cpus:close()
74  end
75  if type(cfg.cores) == "number" then
76  coreMask = cfg.cores
77  cores = {}
78  -- TODO: support more than 32 cores but bit operations on 64 bit types are currently not supported in luajit
79  for i = 0, 31 do
80  if bit.band(coreMask, bit.lshift(1, i)) ~= 0 then
81  cores[#cores + 1] = i
82  end
83  end
84  if cfg.cores >= 2^32 then
85  print("Warning: more than 32 cores are currently not supported in bitmask format, sorry")
86  print("Use a table as a work-around")
87  return
88  end
89  elseif type(cfg.cores) == "table" then
90  cores = cfg.cores
91  coreMask = 0
92  for i, v in ipairs(cores) do
93  coreMask = coreMask + 2^v
94  end
95  end
96  local argv = { "MoonGen" }
97  if cfg.noHugeTlbFs then
98  argv[#argv + 1] = "--no-huge"
99  end
100  argv[#argv + 1] = ("-c0x%X"):format(coreMask)
101  argv[#argv + 1] = "-n" .. (cfg.memoryChannels or 4) -- todo: auto-detect
102  local argc = #argv
103  dpdkc.rte_eal_init(argc, ffi.new("const char*[?]", argc, argv))
104  return true
105 end

138 --- Wait for a task and return any arguments returned by the task
139 function task:wait()
140  checkCore()
141  while true do
142  if dpdkc.rte_eal_get_lcore_state(self.core) ~= dpdkc.RUNNING then
143  -- task is finished
144  local result = dpdkc.get_result(self.id)
145  if result == nil then
146  -- thread crashed :(
147  return
148  end
149  local resultString = ffi.string(result)
150  dpdkc.free(result)
151  return unpackAll(loadstring(resultString)())
152  end
153  ffi.C.usleep(100)
154  end
155 end

182 --- launches the lua file on the first free core
183 function mod.launchLua(...)
184  checkCore()
185  for i = 2, #cores do -- skip master
186  local core = cores[i]
187  local status = dpdkc.rte_eal_get_lcore_state(core)
188  if status == dpdkc.FINISHED then
189  dpdkc.rte_eal_wait_lcore(core)
190  -- should be guaranteed to be in WAIT state now according to DPDK documentation
191  status = dpdkc.rte_eal_get_lcore_state(core)
192  end
193  if status == dpdkc.WAIT then -- core is in WAIT state
194  return mod.launchLuaOnCore(core, ...)
195  end
196  end
197  error("not enough cores to start this lua task")
198 end
