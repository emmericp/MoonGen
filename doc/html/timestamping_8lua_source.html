345  end

398 function mod.syncClocks(dev1, dev2)
399  local regs1 = timeRegisters[dev1:getPciId()]
400  local regs2 = timeRegisters[dev2:getPciId()]
401  if regs1[1] ~= regs2[1] then
402  error("NICs incompatible, cannot sync clocks")
403  end
404  if regs1[2] ~= regs2[2]
405  or regs1[3] ~= regs2[3]
406  or regs1[4] ~= regs2[4]
407  or regs1[5] ~= regs2[5] then
408  error("NYI: NICs use different timestamp registers")
409  end
410  dpdkc.sync_clocks(dev1.id, dev2.id, select(2, unpack(regs1)))
411 end

463 --- Try to measure the latency of a single packet.
464 --- @param pktSize optional, the size of the generated packet, optional, defaults to the smallest possible size
465 --- @param packetModifier optional, a function that is called with the generated packet, e.g. to modified addresses
466 --- @param maxWait optional (cannot be the only argument) the time in ms to wait before the packet is assumed to be lost (default = 15)
467 function timestamper:measureLatency(pktSize, packetModifier, maxWait)
468  if type(pktSize) == "function" then -- optional first argument was skipped
469  return self:measureLatency(nil, pktSize, packetModifier)
470  end
471  pktSize = pktSize or self.udp and 76 or 60
472  maxWait = (maxWait or 15) / 1000
473  self.txBufs:alloc(pktSize)
474  local buf = self.txBufs[1]
475  buf:enableTimestamps()
476  local expectedSeq = self.seq
477  self.seq = (self.seq + 1) % 2^16
478  if self.udp then
479  buf:getUdpPtpPacket().ptp:setSequenceID(expectedSeq)
480  else
481  buf:getPtpPacket().ptp:setSequenceID(expectedSeq)
482  end
483  if packetModifier then
484  packetModifier(buf)
485  end
486  if self.udp then
487  -- change timestamped UDP port as each packet may be on a different port
488  self.rxQueue:enableTimestamps(buf:getUdpPacket().udp:getDstPort())
489  self.txBufs:offloadUdpChecksums()
490  end
491  mod.syncClocks(self.txDev, self.rxDev)
492  -- clear any "leftover" timestamps
493  if self.rxDev:hasTimestamp() then
494  self.rxQueue:getTimestamp()
495  end
496  self.txQueue:send(self.txBufs)
497  local tx = self.txQueue:getTimestamp(500)
498  if tx then
499  -- sent was successful, try to get the packet back (assume that it is lost after a given delay)
500  local timer = timer:new(maxWait)
501  while timer:running() do
502  local rx = self.rxQueue:tryRecv(self.rxBufs, 1000)
503  local timestampedPkt = self.rxDev:hasTimestamp()
504  if not timestampedPkt then
505  -- NIC didn't save a timestamp yet, just throw away the packets
506  self.rxBufs:freeAll()
507  else
508  -- received a timestamped packet (not necessarily in this batch)
509  -- FIXME: this loop may run into an ugly edge-case where we somehow
510  -- lose the timestamped packet during reception (e.g. when this is
511  -- running on a shared core and no filters are set), this case isn't handled here
512  for i = 1, rx do
513  local buf = self.rxBufs[i]
514  local seq = (self.udp and buf:getUdpPtpPacket() or buf:getPtpPacket()).ptp:getSequenceID()
515  -- not sure if checking :hasTimestamp is worth it
516  -- the flag seems to be quite pointless
517  if buf:hasTimestamp() and seq == expectedSeq and seq == timestampedPkt then
518  -- yay!
519  local rxTs = self.rxQueue:getTimestamp()
520  if not rxTs then
521  -- can happen if you hotplug cables
522  return nil
523  end
524  local delay = (rxTs - tx) * self.rxDev:getTimestampScale()
525  self.rxBufs:freeAll()
526  return delay
527  elseif buf:hasTimestamp() and seq == timestampedPkt then
528  -- we got a timestamp but the wrong sequence number. meh.
529  self.rxQueue:getTimestamp() -- clears the register
530  -- continue, we may still get our packet :)
531  elseif seq == expectedSeq and seq ~= timestampedPkt then
532  -- we got our packet back but it wasn't timestamped
533  -- we likely ran into the previous case earlier and cleared the ts register too late
534  self.rxBufs:freeAll()
535  return
536  end
537  end
538  end
539  end
540  -- looks like our packet got lost :(
541  return
542  else
543  -- uhm, how did this happen? an unsupported NIC should throw an error earlier
544  print("Warning: failed to timestamp packet on transmission")
545  timer:new(maxWait):wait()
546  end
547 end
