83 --- Add a number to an IPv4 address.
84 --- Max. 32 bit, commutative.
85 --- @param lhs Address in 'union ip4_address' format.
86 --- @param rhs Number to add (32 bit integer).
87 --- @return Resulting address in uint32 format.
88 function ip4Addr.__add(lhs, rhs)
89  -- calc ip (self) + number (val)
90  local self, val
91  if istype(ip4AddrType, lhs) then
92  self = lhs
93  val = rhs
94  else
95  -- commutative for number + ip
96  self = rhs
97  val = lhs
98  end -- TODO: ip + ip?
99 
100  return self:get() + val
101 end

412 --- Set all members of the ip header.
413 --- Per default, all members are set to default values specified in the respective set function.
414 --- Optional named arguments can be used to set a member to a user-provided value.
415 --- @param args Table of named arguments. Available arguments: Version, HeaderLength, TOS, Length, ID, Flags, Fragment, TTL, Protocol, Checksum, Src, Dst
416 --- @param pre prefix for namedArgs. Default 'ip4'.
417 --- @code
418 --- fill() --- only default values
419 --- fill{ ipSrc="1.1.1.1", ipTTL=100 } --- all members are set to default values with the exception of ipSrc and ipTTL
420 --- @endcode
421 function ip4Header:fill(args, pre)
422  args = args or {}
423  pre = pre or "ip4"
424 
425  self:setVersion(args[pre .. "Version"])
426  self:setHeaderLength(args[pre .. "HeaderLength"])
427  self:setTOS(args[pre .. "TOS"])
428  self:setLength(args[pre .. "Length"])
429  self:setID(args[pre .. "ID"])
430  self:setFlags(args[pre .. "Flags"])
431  self:setFragment(args[pre .. "Fragment"])
432  self:setTTL(args[pre .. "TTL"])
433  self:setProtocol(args[pre .. "Protocol"])
434  self:setChecksum(args[pre .. "Checksum"])
435 
436  local src = pre .. "Src"
437  local dst = pre .. "Dst"
438  args[src] = args[src] or "192.168.1.1"
439  args[dst] = args[dst] or "192.168.1.2"
440 
441  -- if for some reason the address is in 'union ip4_address' format, cope with it
442  if type(args[src]) == "string" then

451  end
452 end

494 --- Resolve which header comes after this one (in a packet).
495 --- For instance: in tcp/udp based on the ports.
496 --- This function must exist and is only used when get/dump is executed on
497 --- an unknown (mbuf not yet casted to e.g. tcpv6 packet) packet (mbuf)
498 --- @return String next header (e.g. 'udp', 'icmp', nil)
499 function ip4Header:resolveNextHeader()
500  local proto = self:getProtocol()
501  for name, _proto in pairs(mapNameProto) do
502  if proto == _proto then
503  return name
504  end
505  end
506  return nil
507 end

509 --- Change the default values for namedArguments (for fill/get).
510 --- This can be used to for instance calculate a length value based on the total packet length.
511 --- See proto/ip4.setDefaultNamedArgs as an example.
512 --- This function must exist and is only used by packet.fill.
513 --- @param pre The prefix used for the namedArgs, e.g. 'ip4'
514 --- @param namedArgs Table of named arguments (see See Also)
515 --- @param nextHeader The header following after this header in a packet
516 --- @param accumulatedLength The so far accumulated length for previous headers in a packet
517 --- @see ip4Header:fill
518 function ip4Header:setDefaultNamedArgs(pre, namedArgs, nextHeader, accumulatedLength)
519  -- set length
520  if not namedArgs[pre .. "Length"] and namedArgs["pktLength"] then
521  namedArgs[pre .. "Length"] = namedArgs["pktLength"] - accumulatedLength
522  end
523 
524  -- set protocol
525  if not namedArgs[pre .. "Protocol"] then
526  for name, type in pairs(mapNameProto) do
527  if nextHeader == name then
528  namedArgs[pre .. "Protocol"] = type
529  break
530  end
531  end
532  end
533  return namedArgs
534 end
