66 --- Iterate over all keys/values in a namespace
67 -- Note: namespaces do not offer a 'normal' iterator (e.g. through a __pair metamethod) due to locking.
68 -- Iterating over a table requires a lock on the whole table; ensuring that the lock is released is
69 -- easier with a forEach method than with a regular iterator.
70 -- @param cb function to call, receives (key, value) as arguments
71 function namespace:forEach(cb)
72  local caughtError
73  C.namespace_iterate(self, function(key, val)
74  if caughtError then
75  return

77  -- avoid throwing an error across the C++ frame unnecessarily
78  -- not sure if this would work properly when compiled with clang instead of gcc
79  local ok, err = xpcall(cb, function(err)

81  end, ffi.string(key), loadstring(ffi.string(val))())
82  if not ok then

84  end
85  end)
