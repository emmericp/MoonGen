68 --- Add a number to an IPv6 address.
69 -- Max. 64bit, commutative.
70 -- @param lhs Address in 'union ip6_address' format.
71 -- @param rhs Number to add (64 bit integer).
72 -- @return Resulting address in 'union ip6_address' format.
73 function ip6Addr.__add(lhs, rhs)
74  -- calc ip (self) + number (val)
75  local self, val
76  if istype(ip6AddrType, lhs) then
77  self = lhs
78  val = rhs
79  else
80  -- commutative for number + ip
81  self = rhs
82  val = lhs
83  end -- TODO: ip + ip?
84  local addr = ip6AddrType()
85  local low, high = self.uint64[0], self.uint64[1]
86  low = low + val
87  -- handle overflow
88  if low < val and val > 0 then
89  high = high + 1
90  -- handle underflow
91  elseif low > -val and val < 0 then
92  high = high - 1
93  end
94  addr.uint64[0] = low
95  addr.uint64[1] = high
96  return addr
97 end
