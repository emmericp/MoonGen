85 --- Add a number to an IPv6 address.
86 --- Max. 64bit, commutative.
87 --- @param lhs Address in 'union ip6_address' format.
88 --- @param rhs Number to add (64 bit integer).
89 --- @return Resulting address in 'union ip6_address' format.
90 function ip6Addr.__add(lhs, rhs)
91  -- calc ip (self) + number (val)
92  local self, val
93  if istype(ip6AddrType, lhs) then
94  self = lhs
95  val = rhs
96  else
97  -- commutative for number + ip
98  self = rhs
99  val = lhs
100  end -- TODO: ip + ip?
101  local addr = ip6AddrType()
102  local low, high = self.uint64[0], self.uint64[1]
103  low = low + val
104  -- handle overflow
105  if low < val and val > 0 then
106  high = high + 1
107  -- handle underflow
108  elseif low > -val and val < 0 then
109  high = high - 1
110  end
111  addr.uint64[0] = low
112  addr.uint64[1] = high
113  return addr
114 end

359 --- Set all members of the ip6 header.
360 --- Per default, all members are set to default values specified in the respective set function.
361 --- Optional named arguments can be used to set a member to a user-provided value.
362 --- @param args Table of named arguments. Available arguments: ip6Version, ip6TrafficClass, ip6FlowLabel, ip6Length, ip6NextHeader, ip6TTL, ip6Src, ip6Dst
363 --- @param pre prefix for namedArgs. Default 'ip6'.
364 --- @code
365 --- fill() --- only default values
366 --- fill{ ip6Src="f880::ab", ip6TTL=101 } --- all members are set to default values with the exception of ip6Src and ip6TTL
367 --- @endcode
368 function ip6Header:fill(args, pre)
369  args = args or {}
370  pre = pre or "ip6"
371 
372  self:setVersion(args[pre .. "Version"])
373  self:setTrafficClass(args[pre .. "TrafficClass"])
374  self:setFlowLabel(args[pre .. "FlowLabel"])
375  self:setLength(args[pre .. "Length"])
376  self:setNextHeader(args[pre .. "NextHeader"])
377  self:setTTL(args[pre .. "TTL"])
378 
379  local src = pre .. "Src"
380  local dst = pre .. "Dst"
381  args[src] = args[src] or "fe80::1"
382  args[dst] = args[dst] or "fe80::2"
383 
384  -- if for some reason the address is in 'union ip6_address' format, cope with it
385  if type(args[src]) == "string" then
386  self:setSrcString(args[src])
387  else
388  self:setSrc(args[src])
389  end
390  if type(args[dst]) == "string" then
391  self:setDstString(args[dst])
392  else
393  self:setDst(args[dst])
394  end
395 end

432 --- Resolve which header comes after this one (in a packet).
433 --- For instance: in tcp/udp based on the ports.
434 --- This function must exist and is only used when get/dump is executed on

447 --- Change the default values for namedArguments (for fill/get).
448 --- This can be used to for instance calculate a length value based on the total packet length.
449 --- See proto/ip4.setDefaultNamedArgs as an example.
450 --- This function must exist and is only used by packet.fill.
451 --- @param pre The prefix used for the namedArgs, e.g. 'ip6'
452 --- @param namedArgs Table of named arguments (see See Also)
453 --- @param nextHeader The header following after this header in a packet
454 --- @param accumulatedLength The so far accumulated length for previous headers in a packet
455 --- @see ip6Header:fill
456 function ip6Header:setDefaultNamedArgs(pre, namedArgs, nextHeader, accumulatedLength)
457  -- set length
458  if not namedArgs[pre .. "Length"] and namedArgs["pktLength"] then
459  namedArgs[pre .. "Length"] = namedArgs["pktLength"] - (accumulatedLength + 40)
460  end
461 
462  -- set protocol
463  if not namedArgs[pre .. "NextHeader"] then
464  for name, type in pairs(mapNameProto) do
465  if nextHeader == name then
466  namedArgs[pre .. "NextHeader"] = type
467  break
468  end
469  end
470  end
471  return namedArgs
472 end
