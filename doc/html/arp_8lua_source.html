268 --- Set all members of the ip header.
269 --- Per default, all members are set to default values specified in the respective set function.
270 --- Optional named arguments can be used to set a member to a user-provided value.
271 --- @param args Table of named arguments. Available arguments: HardwareAddressType, ProtoAddressType, HardwareAddressLength, ProtoAddressLength, Operation, HardwareSrc, HardwareDst, ProtoSrc, ProtoDst
272 --- @param pre prefix for namedArgs. Default 'arp'.
273 --- @code
274 --- fill() --- only default values
275 --- fill{ arpOperation=2, ipTTL=100 } --- all members are set to default values with the exception of arpOperation
276 --- @endcode
277 function arpHeader:fill(args, pre)
278  args = args or {}
279  pre = pre or "arp"
280 
281  self:setHardwareAddressType(args[pre .. "HardwareAddressType"])
282  self:setProtoAddressType(args[pre .. "ProtoAddressType"])
283  self:setHardwareAddressLength(args[pre .. "HardwareAddressLength"])
284  self:setProtoAddressLength(args[pre .. "ProtoAddressLength"])
285  self:setOperation(args[pre .. "Operation"])
286 
287  local hwSrc = pre .. "HardwareSrc"
288  local hwDst = pre .. "HardwareDst"
289  local prSrc = pre .. "ProtoSrc"
290  local prDst = pre .. "ProtoDst"
291  args[hwSrc] = args[hwSrc] or "01:02:03:04:05:06"
292  args[hwDst] = args[hwDst] or "07:08:09:0a:0b:0c"
293  args[prSrc] = args[prSrc] or "0.1.2.3"
294  args[prDst] = args[prDst] or "4.5.6.7"
295 
296  -- if for some reason the address is in 'struct mac_address'/'union ipv4_address' format, cope with it
297  if type(args[hwSrc]) == "string" then
298  self:setHardwareSrcString(args[hwSrc])
299  else
300  self:setHardwareSrc(args[hwSrc])
301  end
302  if type(args[hwDst]) == "string" then
303  self:setHardwareDstString(args[hwDst])
304  else
305  self:setHardwareDst(args[hwDst])
306  end
307 
308  if type(args[prSrc]) == "string" then
309  self:setProtoSrcString(args[prSrc])
310  else
311  self:setProtoSrc(args[prSrc])
312  end
313  if type(args[prDst]) == "string" then
314  self:setProtoDstString(args[prDst])
315  else
316  self:setProtoDst(args[prDst])
317  end
318 end

341 --- Retrieve the values of all members.
342 --- @return Values in string format.
343 function arpHeader:getString()
344  local str = "ARP hrd " .. self:getHardwareAddressTypeString()
345  .. " (hln " .. self:getHardwareAddressLengthString()
346  .. ") pro " .. self:getProtoAddressTypeString()
347  .. " (pln " .. self:getProtoAddressLength(String)
348  .. ") op " .. self:getOperationString()
349 
350  local op = self:getOperation()
351  if op == arp.OP_REQUEST then
352  str = str .. " who-has " .. self:getProtoDstString()
353  .. " (" .. self:getHardwareDstString()
354  .. ") tell " .. self:getProtoSrcString()
355  .. " (" .. self:getHardwareSrcString()
356  .. ")"
357  elseif op == arp.OP_REPLY then
358  str = str .. " " .. self:getProtoSrcString()
359  .. " is-at " .. self:getHardwareSrcString()
360  .. " (for " .. self:getProtoDstString()
361  .. " @ " .. self:getHardwareDstString()
362  .. ")"
363  else
364  str = str .. " " .. self:getHardwareSrcString()
365  .. " > " .. self:getHardwareDstString()
366  .. " " .. self:getProtoSrcString()
367  .. " > " .. self:getProtoDstString()
368  end
369 
370  return str
371 end

407 --- Arp handler task, responds to ARP queries for given IPs and performs arp lookups
408 --- @todo TODO implement garbage collection/refreshing entries \n
409 --- the current implementation does not handle large tables efficiently \n
410 --- @todo TODO multi-NIC support
411 arp.arpTask = "__MG_ARP_TASK"

413 --- Arp table
414 --- TODO docu
415 local arpTable = ns:get()
416 
417 --- Arp handler task
418 --- @param rxQueue The queue that receives arp packets
419 --- @param txQueue The queue that sends arp packets
420 --- @param ips List of ip addresses. The first will be used as source address in ARP requests.
421 --- @todo TODO docu
422 local function arpTask(rxQueue, txQueue, ips)
423  arpTable.taskRunning = true
424  if type(ips) ~= "table" then
425  ips = { ips }
426  end
427  local ipToMac = {}
428  for i, v in ipairs(ips) do
429  if type(v) == "string" then
430  v = parseIPAddress(v)
431  ips[i] = v
432  end
433  ipToMac[v] = true -- TODO: support different MACs for different IPs
434  end
435  if rxQueue.dev ~= txQueue.dev then
436  error("both queues must belong to the same device")
437  end
438  local arpSrcIP = ips[1] -- the source address for ARP requests
439 
440  local dev = rxQueue.dev
441  local devMac = dev:getMac()
442  local rxBufs = memory.createBufArray(1)
443  local txMem = memory.createMemPool(function(buf)
444  buf:getArpPacket():fill{
445  ethSrc = devMac,
446  arpOperation = arp.OP_REPLY,
447  arpHardwareSrc = devMac,
448  arpProtoSrc = devIP,
449  pktLength = 60
450  }
451  end)
452  local txBufs = txMem:bufArray(1)
453  dev:l2Filter(eth.TYPE_ARP, rxQueue)
454 
455  while dpdk.running() do
456  rx = rxQueue:tryRecvIdle(rxBufs, 1000)
457  assert(rx <= 1)
458  if rx > 0 then
459  local rxPkt = rxBufs[1]:getArpPacket()
460  if rxPkt.eth:getType() == eth.TYPE_ARP then
461  if rxPkt.arp:getOperation() == arp.OP_REQUEST then
462  local ip = rxPkt.arp:getProtoDst()
463  local mac = ipToMac[ip]
464  if mac then
465  if mac == true then
466  mac = devMac
467  end
468  txBufs:alloc(60)
469  -- TODO: a single-packet API would be nice for things like this
470  local pkt = txBufs[1]:getArpPacket()
471  pkt.eth:setDst(rxPkt.eth:getSrc())
472  pkt.arp:setOperation(arp.OP_REPLY)
473  pkt.arp:setHardwareDst(rxPkt.arp:getHardwareSrc())
474  pkt.arp:setProtoDst(rxPkt.arp:getProtoSrc())
475  pkt.arp:setProtoSrc(ip)
476  txQueue:send(txBufs)
477  end
478  elseif rxPkt.arp:getOperation() == arp.OP_REPLY then
479  -- learn from all arp replies we see (suspicable to arp cache poisoning but that doesn't matter here)
480  local mac = rxPkt.arp:getHardwareSrcString()
481  local ip = rxPkt.arp:getProtoSrcString()
482  arpTable[tostring(parseIPAddress(ip))] = { mac = mac, timestamp = dpdk.getTime() }
483  end
484  end
485  rxBufs:freeAll()
486  end
487  -- send outstanding requests
488  arpTable:forEach(function(ip, value)
489  -- TODO: refresh or GC old entries
490  if value ~= "pending" then
491  return
492  end
493  arpTable[ip] = "requested"
494  -- TODO: the format should be compatible with parseIPAddress
495  ip = tonumber(ip)
496  txBufs:alloc(60)
497  local pkt = txBufs[1]:getArpPacket()
498  pkt.eth:setDstString(eth.BROADCAST)
499  pkt.arp:setOperation(arp.OP_REQUEST)
500  pkt.arp:setHardwareDstString(eth.BROADCAST)
501  pkt.arp:setProtoDst(ip)
502  pkt.arp:setProtoSrc(arpSrcIP)
503  txQueue:send(txBufs)
504  end)
505  --dpdk.sleepMillisIdle(1)
506  end
507 end

509 --- Perform a lookup in the ARP table.
510 --- @param ip The ip address in string or cdata format to look up.
511 function arp.lookup(ip)
512  if type(ip) == "string" then
513  ip = parseIPAddress(ip)
514  elseif type(ip) == "cdata" then
515  ip = ip:get()
516  end
517  if not arpTable.taskRunning then
518  error("ARP task is not running")
519  end
520  local mac = arpTable[tostring(ip)]
521  if mac and mac ~= "pending" then
522  return mac.mac, mac.timestamp
523  end
524  if mac ~= "requested" then
525  arpTable[tostring(ip)] = "pending" -- FIXME: this needs a lock
526  end
527  return nil
528 end
