42 --- Enable mempool recycling.

53 local function getPoolFromCache(socket, n, bufSize)
54  if not cacheEnabled then
55  return
56  end
57  local pool
58  mempoolCache.lock(function()
59  -- TODO: pass an iterator context to the callback
60  -- the context could then run functions like abort() or removeCurrent()
61  local result
62  mempoolCache:forEach(function(key, pool)
63  if result then
64  return
65  end
66  if pool.socket == socket
67  and pool.n == n
68  and pool.bufSize == bufSize
69  and pool.core == dpdk.getCore() then
70  result = key
71  end
72  end)
73  if result then
74  pool = mempoolCache[result].pool
75  mempoolCache[result] = nil
76  end
77  end)
78  if pool then
79  local bufs = {}
80  for i = 1, n do
81  local buf = pool:alloc(bufSize)
82  ffi.fill(buf.data, buf.len, 0)
83  bufs[#bufs + 1] = buf
84  end
85  for _, v in ipairs(bufs) do
86  dpdkc.rte_pktmbuf_free_export(v)
87  end
88  end
89  return pool
90 end

92 --- Create a new memory pool.
93 -- Memory pools are recycled once the owning task terminates.
94 -- Call :retain() for mempools that are passed to other tasks.
95 -- @param n optional (default = 2047), size of the mempool
96 -- @param func optional, init func, called for each argument
97 -- @param socket optional (default = socket of the calling thread), NUMA association. This cannot be the only argument in the call.
98 -- @param bufSize optional the size of each buffer, can only be used if all other args are passed as well
99 function mod.createMemPool(n, func, socket, bufSize)
100  if type(n) == "function" then -- (func[, socket])
101  socket = func
102  func = n
103  n = nil
104  elseif type(func) == "number" then -- (n[, socket])
105  socket = func
106  func = nil
107  end
108  n = n or 2047
109  socket = socket or select(2, dpdk.getCore())
110  bufSize = bufSize or 2048
111  -- TODO: get cached mempool from the mempool pool if possible and use that instead
112  local mem = getPoolFromCache(socket, n, bufSize) or dpdkc.init_mem(n, socket, bufSize)
113  if func then
114  local bufs = {}
115  for i = 1, n do
116  local buf = mem:alloc(1522)
117  func(buf)
118  bufs[#bufs + 1] = buf
119  end
120  for i, v in ipairs(bufs) do
121  dpdkc.rte_pktmbuf_free_export(v)
122  end
123  end
124  mempools[#mempools + 1] = {
125  pool = mem,
126  socket = socket,
127  n = n,
128  bufSize = bufSize,
129  core = dpdk.getCore()
130  }
131  return mem
132 end
